제이쿼리는 다운받아서 사용 cdn방식은 code.jqeury.com / 다운은 download the compressed, production jqery 3.6.0 -> 다른 이름으로 저장
js폴더만들어서 그 안에 저장
로그인 버튼 클릭시 로그인 기능 동작하도록 ajax 사용
인터페이스 자료 보고 복사 붙여넣기 -> url 맨 앞에 슬래시는 빼고
				요청 파라미터가 data -> 그대로 복사 "ID=" + 사용자가 입력한 아이디 + "&PW=" + 사용자가 입력한 비밀번호,
$.ajax({
	url: "member/login",
	type: "POST",
	data: " "ID=" + 사용자가 입력한 아이디 + "&PW=" + 사용자가 입력한 비밀번호 ",
	success: function(){
		location.href = "로그인 성공했을 때 이동할 페이지 URL";
	},
	error: function(response){
		if(response.status == 400) {
			// 로그인 인터페이스가 400으로 응답한 경우
			alert("아이디 또는 비밀번호를 확인하세요1")
		} else if(response.status == 401) {
			// 로그인 인터페이스가 401으로 응답한 경우
			alert("아이디 또는 비밀번호를 확인하세요2");
		} else if(response.status == 404) {
			// 로그인 인터페이스가 404으로 응답한 경우
			alert("아이디 또는 비밀번호를 확인하세요3");
		}
	}
});
